## 堆排序
```c++
class HeapSort{
public:
    int* heapSort(int* A, int n){
        // 初始化,建堆
        for(int i = n/2-1; i >= 0; i--){
            max_heapify(A, i, n-1);
        }
        // 排序
        for(int i = n-1; i > 0; i--){
            swap(A[0], A[i]);
            max_heapify(A, 0, i-1);
        }
        return A; 
    }

    // 最大堆调整：将堆的末端子节点作调整，使得子节点永远小于父节点
    // start为当前需要调整最大堆的位置，end为调整边界
    void max_heapify(int* A, int start, int end){
        int dad = start;
        int son = 2*dad + 1;
        while(son <= end){
            if(son+1 <= end && A[son] < A[son+1])
                son++;
            if(A[dad] > A[son])
                return ;
            else{
                swap(A[dad], A[son]);
                dad = son;
                son = 2 * dad + 1;
            }
        }
    }
};
```
